#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

int numbers[500501];

int main()
{
	//const int T = 10;
	int T;
	cin >> T;

	int i, j, k, l;
	for (i = 0; i < T; i++) {

		int answer = 0;
		int N;
		cin >> N;

		memset(numbers, 0, sizeof(numbers));
		
		int arr[1001] = { 0 };
		for (j = 0; j < N; j++) {
			cin >> arr[j];
		}

		int index = 0;
		for (j = 1; j <= N; j++) {
			for (k = 0; k <= N - j; k++) {
				string num = "";
				for (l = 0; l < j; l++) {
					num.push_back(arr[k + l] + '0');
				}

				numbers[index++] = atoi(num.c_str());
			}
		}
		
		sort(numbers, numbers + index);
		for (j = 0; j < index - 1; j++) {
			if (numbers[j + 1] - numbers[j] > 1) {
				answer = numbers[j] + 1;
				break;
			}
		}

		cout << "#" << i + 1 << " " << answer << "\n";
	}

	return 0;
}
