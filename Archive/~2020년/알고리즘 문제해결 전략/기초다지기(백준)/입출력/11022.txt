#include<iostream>
#include<string>

using namespace std;

int main()
{
	int T;
	cin >> T;
	getchar();

	int i;
	for(i=0; i<T; i++)
	{
		string str;
		getline(cin, str);

		if (str == "")
		{
			break;
		}
		else
		{
			int sum = (str[0] - '0') + (str[2] - '0');
			cout << "Case #" << i + 1 << ": " << str[0] << " + " << str[2] << " = " << sum << endl;
		}
	}
}
