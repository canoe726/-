#include <string>
#include <vector>
#include <iostream>

using namespace std;

vector<int> solution(int n, vector<string> words) {
	vector<int> answer;

	int cur_index = 1;
	int order = 1;
	int fail = 0;
	int i, j;

	string before = words[0];
	
	for (i = 1; i < words.size(); i++)
	{
		cur_index++;
		if (cur_index > n)
		{
			cur_index = 1;
			order++;
		}

		if (before[before.size() - 1] == words[i][0])
		{
			before = words[i];
			
			// 중복 체크
			for (j = i - 1; j >= 0; j--)
			{
				if (words[i] == words[j])
				{
					fail = 1;
					break;
				}
			}
			if (fail == 1)
			{
				break;
			}
		}
		else
		{
			fail = 1;
			break;
		}
	}

	if (fail == 0)
	{
		cur_index = 0;
		order = 0;
	}
	answer.push_back(cur_index);
	answer.push_back(order);

	return answer;
}

int main()
{
	int n;
	cin >> n;
	
	vector<string> words;
	vector<int> res;
	int size;
	cin >> size;
	int i;
	for (i = 0; i < size; i++)
	{
		string str;
		cin >> str;
		words.push_back(str);
	}

	res = solution(n, words);

	for (i = 0; i < res.size(); i++)
	{
		cout << res[i] << ' ';
	}
	cout << endl;
}
