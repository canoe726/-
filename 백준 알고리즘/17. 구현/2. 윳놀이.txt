#include<iostream>

using namespace std;

int main()
{
	int stick[4] = { 0 };
	int i, j;
	for (i = 0; i < 3; i++)
	{
		int result = 0;

		for (j = 0; j < 4; j++)
		{
			cin >> stick[j];
			result += stick[j];
		}

		// ¹è(0) 4°³
		if (result == 0)
		{
			cout << "D ";
		}
		else if (result == 1)
		{
			cout << "C ";
		}
		else if (result == 2)
		{
			cout << "B ";
		}
		else if (result == 3)
		{
			cout << "A ";
		}
		else if (result == 4)
		{
			cout << "E ";
		}
		cout << '\n';
	}
	
	
	return 0;
}
