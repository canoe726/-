#include <iostream>
#include <cstring>

using namespace std;

int main()
{
	int T;
	cin >> T;

	int i, j;
	for (i = 0; i < T; i++) {

		int N;
		cin >> N;

		int m50000, m10000, m5000, m1000, m500, m100, m50, m10;
		m50000 = m10000 = m5000 = m1000 = m500 = m100 = m50 = m10 = 0;
		
		if ((N / 50000) > 0) {
			m50000 += N / 50000;
			N = N - 50000 * m50000;
		}
		if ((N / 10000) > 0) {
			m10000 += N / 10000;
			N = N - 10000 * m10000;
		}
		if ((N / 5000) > 0) {
			m5000 += N / 5000;
			N = N - 5000 * m5000;
		}
		if ((N / 1000) > 0) {
			m1000 += N / 1000;
			N = N - 1000 * m1000;
		}
		if ((N / 500) > 0) {
			m500 += N / 500;
			N = N - 500 * m500;
		}
		if ((N / 100) > 0) {
			m100 += N / 100;
			N = N - 100 * m100;
		}
		if ((N / 50) > 0) {
			m50 += N / 50;
			N = N - 50 * m50;
		}
		if ((N / 10) > 0) {
			m10 += N / 10;
			N = N - 10 * m10;
		}

		cout << "#" << i + 1 << "\n";
		cout << m50000 << " " << m10000 << " " << m5000 << " " << m1000 << " " << m500 << " " << m100 << " " << m50 << " " << m10 << "\n";
	}
	cout << "\n";

	return 0;
}
